13  import org.apache.maven.plugin.logging.Log;
14  import org.apache.maven.plugins.annotations.Mojo;
15  import org.apache.maven.plugins.annotations.Parameter;
16  import org.apache.maven.reporting.AbstractMavenReport;

19  /**
20   * Runs Doxygen to generate API documentation.
21   */
22  @Mojo(name = "report", requiresProject = true)
23  public class Report extends AbstractMavenReport {
24  	@Override
25  	public String getDescription(Locale locale) {
26  		return "Doxygen API documentation.";
27  	}
28  
29  	@Override
30  	public String getName(Locale locale) {
31  		return "Doxygen";
32  	}
33  
34  	/**
35  	 * Doxygen config file location.
36  	 */
37  	@Parameter(defaultValue = "${basedir}/src/doxygen/doxygen.config")
38  	private File doxygenConf;
39  
40  	/**
41  	 * Doxygen output directory.
42  	 */
43  	@Parameter(defaultValue = "${project.reporting.outputDirectory}/doxygen")
44  	private File outputDirectory;
45  
46  	/**
47  	 * Doxygen INPUT parameter.
48  	 */
49  	@Parameter(defaultValue = "${basedir}/src")
50  	private File input;
51  
52  	/**
53  	 * Doxygen executable name. Default assume it's within the PATH.
54  	 */
55  	@Parameter(defaultValue = "doxygen")
56  	private String executableName;
57  
58  	@Parameter(readonly = true, defaultValue = "${project.build.directory}/doxygen.config")
59  	private File derivedDoxygenConf;
60  
61  	@Override
62  	public String getOutputName() {
63  		return "doxygen/html/index";
64  	}
65  
66  	@Override
67  	public boolean isExternalReport() {
68  		return true;
69  	}
70  
71  	@Override
72  	protected void executeReport(Locale locale) throws MavenReportException {
73  		Log log = getLog();
74  		log.info("derivedDoxygenConf: " + derivedDoxygenConf.getName());
75  		log.info("derivedDoxygenConf: " + derivedDoxygenConf.getName());
76  		deriveConfFile();
77  
78  		try {
79  			ProcessBuilder builder = new ProcessBuilder(executableName, derivedDoxygenConf.getPath());
80  			builder.redirectErrorStream(true);
81  			Process shell = builder.start();
82  
83  			BufferedReader shellReader = new BufferedReader(new InputStreamReader(shell.getInputStream()));
84  
85  			String line = null;
86  			while ((line = shellReader.readLine()) != null) {
87  				log.info(line);
88  			}
89  
90  			int result = shell.waitFor();
91  			if (result != 0)
92  				throw new MavenReportException("Doxygen exited with error code: " + result);
93  
94  		} catch (InterruptedException e) {
95  			throw new MavenReportException("Error running Doxygen", e);
96  		} catch (IOException e) {
97  			throw new MavenReportException("Error running Doxygen", e);
98  		}
99  	}
100 
101 	/**
102 	 * Combines our plugin settings with existing Doxygen config file
103 	 */
104 	private void deriveConfFile() throws MavenReportException {
105 		try {
106 			BufferedWriter derivedWriter = new BufferedWriter(new FileWriter(derivedDoxygenConf));
107 
108 			derivedWriter.write("# Settings generated by " + Report.class.getPackage().getName());
109 			derivedWriter.newLine();
110 			derivedWriter.write("PROJECT_NAME = \"" + getProject().getArtifactId() + "\"");
111 			derivedWriter.newLine();
112 			derivedWriter.write("PROJECT_NUMBER = \"" + getProject().getVersion() + "\"");
113 			derivedWriter.newLine();
114 			derivedWriter.write("OUTPUT_DIRECTORY = \"" + outputDirectory + "\"");
115 			derivedWriter.newLine();
116 			derivedWriter.write("INPUT = \"" + input + "\"");
117 			derivedWriter.newLine();
118 			derivedWriter.write("RECURSIVE = YES");
119 			derivedWriter.newLine();
120 			derivedWriter.write("# End of generated settings");
121 			derivedWriter.newLine();
122 			derivedWriter.newLine();
123 
124 			BufferedReader confReader = new BufferedReader(new FileReader(doxygenConf));
125 			String line = null;
126 			while ((line = confReader.readLine()) != null) {
127 				derivedWriter.write(line);
128 				derivedWriter.newLine();
129 			}
130 
131 			derivedWriter.close();
132 			confReader.close();
133 		} catch (FileNotFoundException e) {
134 			throw new MavenReportException("Doxygen config file not found", e);
135 		} catch (IOException e) {
136 			throw new MavenReportException("Error deriving Doxygen config file", e);
137 		}
138 
139 	}
